          \\

        \end{cases}
      $$

      $$
        2X-3Y=\begin{pmatrix}
          -4 & -2 \\

        \end{pmatrix}, ~~~~ X+2Y=\begin{pmatrix}
          -1 & -3 \\
          4 & -5
        \end{pmatrix}
      $$
      Hint: The answer is 
      $$
        X=\dfrac{1}{7}\begin{pmatrix}
          -11 & -13 \\
          26 & -29
        \end{pmatrix}, ~~~~ Y=\dfrac{1}{7}\begin{cases}
          2 & -4 \\
          1 & -3
        \end{cases}
      $$
